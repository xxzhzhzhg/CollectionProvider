    enum SectionType: String, Hashable {
        case user
        case list
    }

    @Published private var datas: [AnyCollectionSection<SectionType>] = [] {
        didSet {
            coordinator.sections = datas
        }


    lazy var coordinator: CollectionSectionProvider = {
        let cr = CollectionSectionProvider()
        return cr
    }()
    
    private var onScroll: ((UIScrollView) -> Void)?
    
    private lazy var collectionView = UICollectionView(frame: .zero, collectionViewLayout: createLayout()).then {
        $0.register(collectionViewCell.self, forCellWithReuseIdentifier: collectionViewCell.identifier)
        $0.dataSource = coordinator
    }


    let items = convers.map { conver in
        CollectionViewProvider(model: conver).toAny()
    }
    datas.append(AnyCollectionSection(id: .list, items: items))
    collectionView.relodaData()


    struct CollectionViewProvider: CollectionItem {
        let model: Model
        var onSelect: (() -> Void)?

        var cellClass: UICollectionViewCell.Type {
            collectionViewCell.self
        }

        func cellForItem(cell: UICollectionViewCell) {
            guard let cell = cell as? collectionViewCell else { return }
            cell.configure(with: model)
        }
    }
